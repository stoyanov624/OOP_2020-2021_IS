# Ламбди и std::function
## За удобство и за спестяване на писане на много код и излишни декларация от c++11 са имплементирани ламбдите, които представлям кратки функции, които можем да напишем "на момента", а std::function ни улеснява живота с указателите към функции. Отново имате примери долу за по - добро разбиране на синтаксиса и логиката.

``` c++
#include <iostream>
#include <functional>

int incrementElement(int element) {
    return element + 1;
}

void incrementElementRef(int &element) {
    element++;
}

int main() {
    // Ламбди - кратки функции, които можем да напишем без излишна декларация извън мейна
    auto maplambda = [] (int x) -> int { return x + 1; }; 
    // отляво надясно след [] се чете така ще имаме функция, която 
    // ще приема аргумент int -> тази функция ще 
    // връща int {<имплементацията на функцията или тялото й>} 
    std::cout << maplambda(1) << std::endl; // ще изпринтира 2
    
    // с [] може да подаваме стойности от външния свят. 
    // с [] празно казваме, че нищо няма да подаваме.
    // с [&] казваме, че можем да подаваме всичко по референция.
    // с [=] казваме, че можем да подаваме всичко по стойност.
    // можем да сме по специфични и ако имаме променлива "а" и искаме да я ползваме вътре 
    // в ламбда функция може да нанишем [a] или [&a] едното взима а по стойност другото по 
    // референция

    int a = 5;
    auto maplambdaTwo = [a] (int x) -> int{return x + a;};
    
    auto maplambdaThree = [&a] () -> void { a = 10; }; // просто ще променим "а" тъй като 
                                                      //  сме я взели по референция :)

    // auto не се препоръчва да се използва, тук е просто да се види пример за ламбда
    // използвайте std::function както е описано по - долу
    
    std::cout << maplambdaTwo(1) <<  std::endl; // ще изпринтира 6;
    maplambdaThree(); // "а" ще стане 6
    std::cout << a << std::endl; // доказателство

    // std::function - може директно да си декларираме указатели към функции като избягваме 
    // използването на using и тем подобни. използваме библиотеката <functional>

    int b = 10;
    int c = 20;
    std::function<int(int)> func1 = incrementElement; 
    // пишем си std::function<<типът>(<аргументите>)>
    b = func1(b); // увеличаваме b с 1
    std::cout << b << std::endl; // доказателство

    std::function<void(int&)> func2 = incrementElementRef;
    func2(c); // увеличаваме 'c' с 1 тук аргумента ни е по референция.
    std::cout << c << std::endl; // доказателство

    // можем да насочваме указател към ламбди :)
    std::function<void()> func3 = [] () -> void {
                                                for(int i = 1; i <= 10; i++) {
                                                    std::cout << i << " ";
                                                    }
                                                }; // ламбдите може и да са по - дълги.
    func3(); // ще изпринтира числата от 1 до 10.

}
```
